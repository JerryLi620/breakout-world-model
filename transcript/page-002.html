"""
models.py – Latent world model for Atari Breakout.

Two decoder modes are supported:

    decoder_type="residual" (default, for one-step overfit debugging):
        delta = tanh(decoder(z))
        x_hat = clamp(x_last + residual_scale * delta, 0, 1)

    decoder_type="erase_draw" (experimental):
        erase_mask = sigmoid(erase_logits)
        draw_mask  = sigmoid(draw_mask_logits)
        draw_value = sigmoid(draw_value_logits)
        x_hat = clamp(x_last * (1 - erase_mask) + draw_mask * draw_value, 0, 1)

The encoder is a CNN over the stacked input frames. The dynamics module is a
GRU on the latent (recurrent). For teacher-forced overfit debugging it is fine

def add_coord_channels(x: torch.Tensor) -> torch.Tensor:
    if x.dim() != 4:
        raise ValueError(f"Expected x with shape (B, C, H, W), got {x.shape}")

    B, C, H, W = x.shape
    device = x.device
    dtype = x.dtype

    y_coords = torch.linspace(-1.0, 1.0, H, device=device, dtype=dtype)
    y_coords = y_coords.view(1, 1, H, 1).expand(B, 1, H, W)

    x_coords = torch.linspace(-1.0, 1.0, W, device=device, dtype=dtype)
    x_coords = x_coords.view(1, 1, 1, W).expand(B, 1, H, W)

    return torch.cat([x, y_coords, x_coords], dim=1)

class Encoder(nn.Module):
    def __init__(self, in_channels: int = 4, latent_dim: int = 256):
        super().__init__()

        self.conv = nn.Sequential(
            nn.Conv2d(in_channels + 2, 32, 4, stride=2, padding=1),
            nn.GroupNorm(8, 32),
            nn.SiLU(),

            nn.Conv2d(32, 64, 4, stride=2, padding=1),
            nn.GroupNorm(8, 64),
            nn.SiLU(),

            nn.Conv2d(64, 128, 4, stride=2, padding=1),
            nn.GroupNorm(8, 128),
            nn.SiLU(),

            nn.Conv2d(128, 256, 4, stride=2, padding=1),
            nn.GroupNorm(16, 256),
            nn.SiLU(),
        )

        self.fc = nn.Sequential(
            nn.Flatten(),
            nn.Linear(256 * 4 * 4, 512),
            nn.SiLU(),
            nn.Linear(512, latent_dim),
            nn.LayerNorm(latent_dim),
        )

    def forward(self, x: torch.Tensor):
        x = add_coord_channels(x)
        h = self.conv(x)
        z = self.fc(h)
        skips = None
        return z, skips

class RecurrentDynamicsModel(nn.Module):
    def __init__(
        self,
        latent_dim: int = 256,
        hidden_dim: int = 512,
        action_dim: int = 0,
        noise_std: float = 0.0,
    ):
        super().__init__()

        self.latent_dim = latent_dim
        self.hidden_dim = hidden_dim
        self.action_dim = action_dim
        self.noise_std = noise_std

        input_dim = latent_dim + action_dim

        self.input_norm = nn.LayerNorm(input_dim)
        self.gru = nn.GRUCell(input_dim, hidden_dim)

        self.to_z = nn.Sequential(
            nn.LayerNorm(hidden_dim),
            nn.Linear(hidden_dim, hidden_dim),
            nn.SiLU(),
            nn.Linear(hidden_dim, latent_dim),
            nn.LayerNorm(latent_dim),
        )

    def init_hidden(
        self,
        batch_size: int,
        device: torch.device,
        dtype: torch.dtype = torch.float32,
    ) -> torch.Tensor:
        return torch.zeros(batch_size, self.hidden_dim, device=device, dtype=dtype)

    def forward(
        self,
        z: torch.Tensor,
        action: torch.Tensor = None,
        h: torch.Tensor = None,
        deterministic: bool = False,
    ):
        if z.dim() != 2:
            raise ValueError(f"Expected z with shape (B, latent_dim), got {z.shape}")

        B = z.size(0)

        if h is None:
            h = self.init_hidden(B, z.device, z.dtype)

        if self.action_dim > 0:
            if action is None:
                raise ValueError("Action must be provided when action_dim > 0.")

            if action.dim() != 2:
                raise ValueError(f"Expected action shape (B, action_dim), got {action.shape}")

            if action.size(1) != self.action_dim:
                raise ValueError(
                    f"Expected action_dim={self.action_dim}, got action shape {action.shape}"
                )

            x = torch.cat([z, action.to(dtype=z.dtype)], dim=-1)
        else:
            x = z

        x = self.input_norm(x)
        h_next = self.gru(x, h)
        z_next = self.to_z(h_next)

        if self.noise_std > 0.0 and not deterministic:
            z_next = z_next + torch.randn_like(z_next) * self.noise_std

        return z_next, h_next

    def __init__(self, latent_dim: int):

class ResidualDecoder(nn.Module):
    """
    Predicts a bounded residual delta to add to x_last.

        delta = tanh(conv(feat))
        x_hat = clamp(x_last + residual_scale * delta, 0, 1)
    """

    def __init__(self, latent_dim: int = 256):
        super().__init__()
        self.body = _DecoderBody(latent_dim)
        self.delta_head = nn.Conv2d(32, 1, kernel_size=3, padding=1)

        # Start near zero residual.
        nn.init.zeros_(self.delta_head.weight)
        nn.init.zeros_(self.delta_head.bias)

    def forward(self, z: torch.Tensor):
        if z.dim() != 2:
            raise ValueError(f"Expected z shape (B, latent_dim), got {z.shape}")
        feat = self.body(z)
        delta = torch.tanh(self.delta_head(feat))
        return {"delta": delta}

class WorldModel(nn.Module):
    def __init__(
        self,
        in_channels: int = 4,
        out_channels: int = 1,
        action_dim: int = 0,
        latent_dim: int = 256,
        hidden_dim: int = 512,
        noise_std: float = 0.0,
        decoder_type: str = "residual",
        residual_scale: float = 0.25,
    ):
        super().__init__()

        if out_channels != 1:
            raise ValueError("Decoder currently assumes grayscale out_channels=1.")

        if decoder_type not in ("residual", "erase_draw"):
            raise ValueError(
                f"decoder_type must be 'residual' or 'erase_draw', got {decoder_type}"
            )

        self.encoder = Encoder(
            in_channels=in_channels,
            latent_dim=latent_dim,
        )

        self.dynamics = RecurrentDynamicsModel(
            latent_dim=latent_dim,
            hidden_dim=hidden_dim,
            action_dim=action_dim,
            noise_std=noise_std,
        )

        if decoder_type == "residual":
            self.decoder = ResidualDecoder(latent_dim=latent_dim)
        else:
            self.decoder = EraseDrawDecoder(latent_dim=latent_dim)

        self.in_channels = in_channels
        self.out_channels = out_channels
        self.action_dim = action_dim
        self.latent_dim = latent_dim
        self.hidden_dim = hidden_dim
        self.noise_std = noise_std
        self.decoder_type = decoder_type
        self.residual_scale = residual_scale

    def init_hidden(
        self,
        batch_size: int,
        device: torch.device,
        dtype: torch.dtype = torch.float32,
    ) -> torch.Tensor:
        return self.dynamics.init_hidden(batch_size, device, dtype)

    def encode(self, x: torch.Tensor):
        return self.encoder(x)

    def predict_latent(self, z, action=None, h=None, deterministic=False):
        return self.dynamics(z=z, action=action, h=h, deterministic=deterministic)

    def decode(self, z: torch.Tensor):
        return self.decoder(z)

    def apply_residual(self, x_last: torch.Tensor, decoder_out: dict):
        delta = decoder_out["delta"]
        x_hat = torch.clamp(x_last + self.residual_scale * delta, 0.0, 1.0)
        return x_hat, {"delta": delta}

    def apply_erase_draw(self, x_last: torch.Tensor, decoder_out: dict, edit_scale: float = 1.0):
        erase_mask = torch.sigmoid(decoder_out["erase_logits"]) * edit_scale
        draw_mask = torch.sigmoid(decoder_out["draw_mask_logits"]) * edit_scale
        draw_value = torch.sigmoid(decoder_out["draw_value_logits"])

        erase_mask = erase_mask.clamp(0.0, 1.0)
        draw_mask = draw_mask.clamp(0.0, 1.0)

        x_erased = x_last * (1.0 - erase_mask)
        x_drawn = x_erased + draw_mask * draw_value
        x_hat = torch.clamp(x_drawn, 0.0, 1.0)

        return x_hat, {
            "erase_mask": erase_mask,
            "draw_mask": draw_mask,
            "draw_value": draw_value,
        }

    def forward(
        self,
        x_t: torch.Tensor,
        action: torch.Tensor = None,
        h: torch.Tensor = None,
        deterministic: bool = False,
        edit_scale: float = 1.0,
        return_aux: bool = False,
    ):
        z_t, _ = self.encode(x_t)

        z_next, h_next = self.predict_latent(
            z=z_t,
            action=action,
            h=h,
            deterministic=deterministic,
        )

        decoder_out = self.decode(z_next)
        x_last = x_t[:, -1:].contiguous()

        if self.decoder_type == "residual":
            x_hat, extras = self.apply_residual(x_last, decoder_out)
        else:
            x_hat, extras = self.apply_erase_draw(x_last, decoder_out, edit_scale=edit_scale)

        if return_aux:
            aux = {**decoder_out, **extras}
            return x_hat, z_t, z_next, h_next, aux

        return x_hat, z_t, z_next, h_next

def effective_rollout_weight(
    global_step: int,
    target_weight: float,
    warmup_steps: int,
    ramp_steps: int,
) -> float:
    if target_weight <= 0.0:
        return 0.0
    if global_step < warmup_steps:
        return 0.0
    if ramp_steps <= 0:
        return target_weight
    progress = (global_step - warmup_steps) / float(ramp_steps)
    progress = max(0.0, min(1.0, progress))
    return target_weight * progress

def simple_one_step_loss(
    pred: torch.Tensor,
    target: torch.Tensor,
    prev_frame: torch.Tensor,
    object_weight: float,
    change_weight: float,
    delta_weight: float,
    false_motion_weight: float,
    object_threshold: float,
    change_threshold: float,
) -> Tuple[torch.Tensor, dict]:
    """
    Clean 3-term loss for one-step overfit debugging:

        1. weighted image loss
        2. changed-pixel delta loss
        3. false-motion loss (weak)
    """
    if pred.shape != target.shape or prev_frame.shape != target.shape:
        raise ValueError(
            f"shape mismatch: pred {pred.shape}, target {target.shape}, prev {prev_frame.shape}"
        )

    object_mask = (target > object_threshold).float()
    change_mask = ((target - prev_frame).abs() > change_threshold).float()

    per_pixel = F.smooth_l1_loss(pred, target, beta=0.01, reduction="none")
    weight = 1.0 + object_weight * object_mask + change_weight * change_mask
    image_loss = (weight * per_pixel).mean()

    pred_delta = pred - prev_frame
    target_delta = target - prev_frame
    delta_error = F.smooth_l1_loss(pred_delta, target_delta, beta=0.01, reduction="none")

    changed_pixels = change_mask.sum().clamp_min(1.0)
    changed_delta_loss = (delta_error * change_mask).sum() / changed_pixels

    false_motion_loss = (delta_error * (1.0 - change_mask)).mean()

    total = (
        image_loss
        + delta_weight * changed_delta_loss
        + false_motion_weight * false_motion_loss
    )

    stats = {
        "image_loss": image_loss.detach(),
        "changed_delta_loss": changed_delta_loss.detach(),
        "false_motion_loss": false_motion_loss.detach(),
        "change_pixels": change_mask.sum().detach(),
    }
    return total, stats


def structured_loss(
    pred: torch.Tensor,
    target: torch.Tensor,
    prev_frame: torch.Tensor,
    aux: dict,
    args: argparse.Namespace,
) -> Tuple[torch.Tensor, dict]:
    """
    Heavier loss with false-disappearance and erase/draw aux terms.
    Only used when --no_simple_one_step is passed AND decoder_type == "erase_draw".
    """
    total, stats = simple_one_step_loss(
        pred=pred,
        target=target,
        prev_frame=prev_frame,
        object_weight=args.object_weight,
        change_weight=args.change_weight,
        delta_weight=args.delta_weight,
        false_motion_weight=args.false_motion_weight,
        object_threshold=args.object_threshold,
        change_threshold=args.change_threshold,
    )

    change_mask = ((target - prev_frame).abs() > args.change_threshold).float()
    true_erase_mask = (
        (prev_frame > args.object_threshold)
        & ((prev_frame - target) > args.change_threshold)
    ).float()
    true_draw_mask = (
        (target > args.object_threshold)
        & ((target - prev_frame) > args.change_threshold)
    ).float()
    should_remain_mask = (
        (prev_frame > args.object_threshold) & (true_erase_mask < 0.5)
    ).float()

    if args.false_disappear_weight > 0.0:
        false_disappear_amount = F.relu(prev_frame - pred - args.change_threshold)
        remain_pixels = should_remain_mask.sum().clamp_min(1.0)
        false_disappear_loss = (
            false_disappear_amount * should_remain_mask
        ).sum() / remain_pixels
        total = total + args.false_disappear_weight * false_disappear_loss
        stats["false_disappear_loss"] = false_disappear_loss.detach()

    if "erase_logits" in aux and (
        args.erase_aux_weight > 0.0 or args.draw_aux_weight > 0.0
    ):
        erase_aux = F.binary_cross_entropy_with_logits(aux["erase_logits"], true_erase_mask)
        draw_aux = F.binary_cross_entropy_with_logits(aux["draw_mask_logits"], true_draw_mask)
        total = total + args.erase_aux_weight * erase_aux + args.draw_aux_weight * draw_aux
        stats["erase_aux_loss"] = erase_aux.detach()
        stats["draw_aux_loss"] = draw_aux.detach()

    if "erase_mask" in aux and args.edit_l1_weight > 0.0:
        edit_l1 = aux["erase_mask"].mean() + aux["draw_mask"].mean()
        total = total + args.edit_l1_weight * edit_l1
        stats["edit_l1"] = edit_l1.detach()

    return total, stats


def compute_step_loss(pred, target, prev_frame, aux, args):
